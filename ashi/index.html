// ============================================================================ // SinoMed content-gate（Storage 多模块版）—— 服务端付费内容按需下发 // ---------------------------------------------------------------------------- // 支持 internal（内科病）和 ashi（阿是肌肉核心字段）两个模块。 // 数据从 Supabase Storage 私有桶 `gate` 读取： // internal → gate/paid_diseases.json （{ [sym]: {t,z} }） // ashi → gate/ashi_paid.json （{ [muscleId]: {核心字段...} }） // // 请求：POST /functions/v1/content-gate // Header: Authorization: Bearer // Body: { "module_id":"internal"|"ashi", "sym":"acidreflux"|"upper_trapezius" } // 或批量 { "module_id":"ashi", "syms":["upper_trapezius","scm"] } // 返回：已购买→200{ok:true,data:{[id]:...}}；未登录→401；未购买→403；其他→400·500 // ============================================================================ import { createClient } from "https://esm.sh/@supabase/supabase-js@2"; const SUPABASE_URL = Deno.env.get("SUPABASE_URL")!; const SERVICE_ROLE = Deno.env.get("SUPABASE_SERVICE_ROLE_KEY")!; const ALLOWED_ORIGINS = (Deno.env.get("GATE_ALLOWED_ORIGINS") ?? "https://sinomed.ca") .split(",").map((s) => s.trim()).filter(Boolean); const admin = createClient(SUPABASE_URL, SERVICE_ROLE, { auth: { autoRefreshToken: false, persistSession: false }, }); // 每个模块的 Storage 数据文件 + 允许的购买 module_id const MODULES: Record = { internal: { file: "paid_diseases.json", grants: ["internal", "allaccess"] }, ashi: { file: "ashi_paid.json", grants: ["ashi", "allaccess"] }, }; // 进程内缓存：module_id -> { data, ids } const CACHE: Record; ids: Set }> = {}; async function loadModule(mid: string): Promise { if (CACHE[mid]) return true; const conf = MODULES[mid]; if (!conf) return false; try { const { data, error } = await admin.storage.from("gate").download(conf.file); if (error || !data) { console.error("[content-gate] 读取失败", conf.file, error); return false; } const obj = JSON.parse(await data.text()); CACHE[mid] = { data: obj, ids: new Set(Object.keys(obj)) }; return true; } catch (e) { console.error("[content-gate] 解析失败", conf.file, e); return false; } } function corsHeaders(origin: string | null) { const allow = origin && ALLOWED_ORIGINS.includes(origin) ? origin : ALLOWED_ORIGINS[0]; return { "Access-Control-Allow-Origin": allow, "Access-Control-Allow-Headers": "authorization, content-type", "Access-Control-Allow-Methods": "POST, OPTIONS", "Vary": "Origin", }; } function json(body: unknown, status: number, origin: string | null) { return new Response(JSON.stringify(body), { status, headers: { "Content-Type": "application/json", ...corsHeaders(origin) }, }); } Deno.serve(async (req) => { const origin = req.headers.get("origin"); if (req.method === "OPTIONS") return new Response("ok", { headers: corsHeaders(origin) }); if (req.method !== "POST") return json({ ok: false, reason: "method_not_allowed" }, 405, origin); const auth = req.headers.get("Authorization") ?? ""; const token = auth.startsWith("Bearer ") ? auth.slice(7) : ""; if (!token) return json({ ok: false, reason: "unauthenticated" }, 401, origin); let module_id = ""; let syms: string[] = []; try { const b = await req.json(); if (typeof b.module_id === "string") module_id = b.module_id; if (typeof b.sym === "string") syms = [b.sym]; else if (Array.isArray(b.syms)) syms = b.syms.filter((x: unknown) => typeof x === "string"); } catch { return json({ ok: false, reason: "bad_request" }, 400, origin); } const conf = MODULES[module_id]; if (!conf) return json({ ok: false, reason: "unknown_module" }, 400, origin); if (syms.length === 0 || syms.length > 80) return json({ ok: false, reason: "bad_request" }, 400, origin); if (!(await loadModule(module_id))) return json({ ok: false, reason: "server_error" }, 500, origin); const bank = CACHE[module_id]; const wanted = syms.filter((s) => bank.ids.has(s)); if (wanted.length === 0) return json({ ok: false, reason: "no_valid_ids" }, 400, origin); const { data: userData, error: userErr } = await admin.auth.getUser(token); if (userErr || !userData?.user) return json({ ok: false, reason: "unauthenticated" }, 401, origin); const userId = userData.user.id; const { data: purchases, error: pErr } = await admin .from("purchases").select("module_id").eq("user_id", userId).in("module_id", conf.grants); if (pErr) return json({ ok: false, reason: "server_error" }, 500, origin); if (!purchases || purchases.length === 0) return json({ ok: false, reason: "not_purchased" }, 403, origin); const out: Record = {}; for (const s of wanted) out[s] = bank.data[s]; return json({ ok: true, data: out }, 200, origin); });
